Combine RVU Data

vctrs::vec_rbind(
  rvu_22.2,
  rvu_22.3,
  rvu_22.4,
  rvu_23.1,
  rvu_23.2,
  rvu_23.3,
  rvu_24.1
  ) |> 
  dplyr::mutate(
    tot_fac_rvu = work_rvu + facility_pe_rvu + mp_rvu,
    tot_non_rvu = work_rvu + non_fac_pe_rvu + mp_rvu,
    par_non_amt = tot_non_rvu * conv_factor,
    par_fac_amt = tot_fac_rvu * conv_factor
  ) |> 
  dplyr::filter(par_fac_amt > 0) |> 
  gt::gt(
    rowname_col = "year", 
    groupname_col = "hcpcs",
    row_group_as_column = TRUE
    ) |> 
  gt::tab_options(
    quarto.disable_processing = TRUE
  )

## Combine RVU Data

```{r}
vctrs::vec_rbind(
  rvu_22.2,
  rvu_22.3,
  rvu_22.4,
  rvu_23.1,
  rvu_23.2,
  rvu_23.3,
  rvu_24.1
  ) |> 
  dplyr::mutate(
    tot_fac_rvu = work_rvu + facility_pe_rvu + mp_rvu,
    tot_non_rvu = work_rvu + non_fac_pe_rvu + mp_rvu,
    par_non_amt = tot_non_rvu * conv_factor,
    par_fac_amt = tot_fac_rvu * conv_factor
  ) |> 
  dplyr::filter(par_fac_amt > 0) |> 
  gt::gt(
    rowname_col = "year", 
    groupname_col = "hcpcs",
    row_group_as_column = TRUE
    ) |> 
  gt::tab_options(
    quarto.disable_processing = TRUE
  )
```
