split_lengths <- function(x, verbose = TRUE) {
  
  stopifnot(is.character(x))
  
  x <- stringfish::sf_gsub(x, "\\*|\\s", "", nthreads = 4L) |> 
    unique_narm() |> 
    stringr::str_sort()
  
  l <- collapse::vlengths(x)
  
  out <- list(
    x1 = x[l == 1],
    x2 = x[l == 2],
    x3 = x[l == 3],
    x4 = x[l == 4],
    x5 = x[l == 5])
  
  if (verbose) {
    
    cli::cat_boxx(
      label = view(out), 
      header = paste0(paste0(rep(cli::symbol$stop, 2), collapse = ""), " Split By Lengths"),
      align = "left", 
      padding = c(0, 1, 0, 1))
    
    # cat("*  Split By Lengths:", sep = "\n")
    # cat(view(out), sep = "\n")  
    return(invisible(out))
    
  } else {
    return(out)
  }
}

lump <- function(x, threshold = 3) {
  
  stopifnot(is.numeric(x))

  xo <- order(x)

  xs <- x[xo]
  
  dlag <- abs(c(0, xs[-1] - xs[seq_along(xs) - 1]))

  bi <- ifelse(dlag >= threshold, 1, 0)

  id <- cumsum(bi) + 1
  
  id[xo]
}

group_hcpcs_3 <- function(x) {

  x <- collapse::get_elem(x, "x3")
  
  if (empty(x)) return(character(0))

  base <- data.table::data.table(
    code = x,
    a1 = take_at(x, 1),
    a2 = take_at(x, 2),
    a3 = take_at(x, 3))

  indices <- base |>
    dplyr::mutate(i1 = dplyr::consecutive_id(a1)) |>
    dplyr::mutate(i2 = dplyr::consecutive_id(a2), .by = a1) |>
    dplyr::mutate(i3 = dplyr::consecutive_id(a3), .by = c(a1, a2))

  orphans <- indices |>
    dplyr::add_count(i1, name = "n1", sort = TRUE) |>
    dplyr::filter((i2 + i3 + n1) == 3) |>
    dplyr::pull(code)

  last <- indices |>
    dplyr::filter(!code %in% orphans) |>
    dplyr::select(code, a1:a3) |>
    fuimus::combine(group_id, columns = c("a1"), sep = "")

  last <- dplyr::left_join(
    last |> dplyr::count(group_id, a2),
    last |> dplyr::count(group_id, name = "g"),
    by = dplyr::join_by(group_id)) |>
    dplyr::filter(n == g) |>
    dplyr::right_join(last, by = dplyr::join_by(group_id, a2)) |>
    dplyr::filter(!is.na(n))

  rest <- indices |>
    dplyr::filter(!code %in% c(orphans, dplyr::pull(last, code))) |>
    fuimus::combine(group_id, columns = c("a1", "a2"), sep = "")

  grouped <- vctrs::vec_c(
    chop(last$code, last$group_id),
    chop(rest$code, rest$group_id)
  )
  
  o <- unlist(grouped[which(collapse::vlengths(grouped) == 1)])
  
  out <- vctrs::vec_c(
    set_names(as.list(orphans), rep("o", length(orphans))),
    set_names(chop(o, dplyr::consecutive_id(take_at(o, 1))), rep("r", length(chop(o, dplyr::consecutive_id(take_at(o, 1)))))),
    set_names(grouped[which(collapse::vlengths(grouped) != 1)], rep("f", length(grouped[which(collapse::vlengths(grouped) != 1)])))
  )
  
  out[collapse::radixorderv(collapse::vlengths(out), sort = TRUE)]
  
}

reduce_runs <- function(x) {
  
  ad <- c(LETTERS, 0:9)
  
  vec  <- rlang::set_names(rep(0, 36), ad)
  
  vec2 <- vec[fuimus::splitter(x)]
  
  vec2 <- vec2[!is.na(vec2)]
  
  vec[names(vec2)] <- 1

  vec_group <- dplyr::tibble(
    value = names(vec),
    key = vec,
    idx = seq_along(vec),
    group = dplyr::consecutive_id(key)) |>
    dplyr::mutate(group_size = dplyr::n(), .by = group) |>
    dplyr::filter(key == 1, group_size >= 3) |>
    dplyr::select(value, group)

  if (empty(vec_group)) return(x)
  
  xgroups <- chop(vec_group$value, vec_group$group) |>
    purrr::map(fuimus::collapser) |>
    purrr::list_c()
  
  if (all(xgroups == paste0(ad, collapse = ""))) return("[A-Z0-9]")
  
  replacements <- dplyr::left_join(
    dplyr::slice_min(vec_group, by = group, order_by = value) |> dplyr::rename(start = value),
    dplyr::slice_max(vec_group, by = group, order_by = value) |> dplyr::rename(end = value),
    by = dplyr::join_by(group)) |>
    glue::glue_data("{start}-{end}") |>
    as.vector()

  res <- stringi::stri_replace_all_regex(x, xgroups, replacements, vectorize_all = FALSE)

  paste0("[", res, "]")
}

    stringfish::sf_grepl(

}

lump <- function(x, threshold = 3) {
  
  stopifnot(is.numeric(x))

  xo <- order(x)

  xs <- x[xo]
  
  dlag <- abs(c(0, xs[-1] - xs[seq_along(xs) - 1]))

  bi <- ifelse(dlag >= threshold, 1, 0)

  id <- cumsum(bi) + 1
  
  id[xo]
}

reduce_runs <- function(x) {
  
  ad <- c(LETTERS, 0:9)
  
  vec  <- rlang::set_names(rep(0, 36), ad)
  
  vec2 <- vec[fuimus::splitter(x)]
  
  vec2 <- vec2[!is.na(vec2)]
  
  vec[names(vec2)] <- 1

  vec_group <- dplyr::tibble(
    value = names(vec),
    key = vec,
    idx = seq_along(vec),
    group = dplyr::consecutive_id(key)) |>
    dplyr::mutate(group_size = dplyr::n(), .by = group) |>
    dplyr::filter(key == 1, group_size >= 3) |>
    dplyr::select(value, group)

  if (empty(vec_group)) return(x)
  
  xgroups <- chop(vec_group$value, vec_group$group) |>
    purrr::map(fuimus::collapser) |>
    purrr::list_c()
  
  if (all(xgroups == paste0(ad, collapse = ""))) return("[A-Z0-9]")
  
  replacements <- dplyr::left_join(
    dplyr::slice_min(vec_group, by = group, order_by = value) |> dplyr::rename(start = value),
    dplyr::slice_max(vec_group, by = group, order_by = value) |> dplyr::rename(end = value),
    by = dplyr::join_by(group)) |>
    glue::glue_data("{start}-{end}") |>
    as.vector()

  res <- stringi::stri_replace_all_regex(x, xgroups, replacements, vectorize_all = FALSE)

  paste0("[", res, "]")
}

p2 <- function(x) {
  
  if (empty(x)) return(character(0))
  
  vec <- stringr::str_split_fixed(x, "", max_vlength(x)) |> 
    as.data.frame() |> 
    purrr::map(unique_narm) |> 
    purrr::map(sort_order) |>
    purrr::map(reduce_runs)
  
  multi <- unname(nchar(vec) > 1)
  
  nobrack <- stringr::str_detect(vec[multi], "\\[|\\]", TRUE)
  
  vec[multi] <- ifelse(any(nobrack), purrr::map_chr(vec[multi], \(x) paste0("[", x, "]")), vec[multi])

  paste0(purrr::list_c(vec), collapse = "")
}

vec_to_regex <- function(x, n) {
  
  if (empty(x)) return(character(0))
  
  postfix <- ifelse(
    stringfish::sf_grepl(
      x, "^\\[?[A-Z]\\-?", nthreads = 4L), 
    "[0-9]", "[A-Z0-9]")
  
  x <- data.table::fcase(
    n == 0, paste0("^", x, "$"),                
    n == 1, paste0("^", x, postfix, "$"),
    default = paste0("^", x, postfix, "{", n, "}$"))
    
  paste0(x , collapse = "")
}
