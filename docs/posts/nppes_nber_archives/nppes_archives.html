make_case <- function(x, y) {
  fcase(
    not_na(x) & not_na(y) & x == y, str_glue("{y}"), 
    not_na(x) & not_na(y) & x != y, str_glue("{x}, {y}"), 
    not_na(x) & na(y), str_glue("{x}"), 
    na(x) & not_na(y), str_glue("{y}")
  )
}

make_zip <- function(x) {
  map_chr(
    x, 
    function(x) {
      if (not_na(x) & nchar(x) == 9) 
        str_glue(
          "{substr(x, 1, 5)}-{substr(x, 6, 9)}"
          )
      else 
        x
    }
  )
}

make_phone <- function(x) {
  
  cnd1 <- "({substr(x, 1, 3)}) {substr(x, 4, 6)}-{substr(x, 7, 10)}"
  cnd2 <- smush(cnd1, ", ({substr(x, 13, 15)})-{substr(x, 16, 18)}-{substr(x, 19, 22)}")
  
  map_chr(x, function(x) {if (not_na(x)) {
    fcase(
      nchar(x) == 10, str_glue(cnd1), 
      nchar(x) > 10, str_glue(cnd2),
      default = str_glue("{x}"))} else str_glue("{x}") })
}

make_case <- function(x, y) {
  fcase(
    not_na(x) & not_na(y) & x == y, str_glue("{y}"), 
    not_na(x) & not_na(y) & x != y, str_glue("{x}, {y}"), 
    not_na(x) & na(y), str_glue("{x}"), 
    na(x) & not_na(y), str_glue("{y}")
  )
}

make_zip <- function(x) {
  map_chr(
    x, 
    function(x) {
      if (not_na(x) & nchar(x) == 9) 
        str_glue(
          "{substr(x, 1, 5)}-{substr(x, 6, 9)}"
          )
      else 
        x
    }
  )
}

make_phone <- function(x) {
  
  cnd1 <- "({substr(x, 1, 3)}) {substr(x, 4, 6)}-{substr(x, 7, 10)}"
  cnd2 <- smush(cnd1, ", ({substr(x, 13, 15)})-{substr(x, 16, 18)}-{substr(x, 19, 22)}")
  
  map_chr(x, function(x) {if (not_na(x)) {
    fcase(
      nchar(x) == 10, str_glue(cnd1), 
      nchar(x) > 10, str_glue(cnd2),
      default = str_glue("{x}"))} else str_glue("{x}") })
}
