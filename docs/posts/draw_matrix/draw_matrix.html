1.1 Lookup

look <- \(code, system) {
  
  constants <- system$c
  possible  <- system$p
  
  names(possible) <- constants
  
  unname(possible[splitter(sf_toupper(code))])
}

look("0G9000Z", pcs)
[1]  1 17 10  1  1  1 34
look("99213", hcpcs)
[1] 10 10  3  2  4

1.2 Make Matrix

medtrix <- \(code, system) {
  
  constants <- system$c
  possible  <- system$p
  axis      <- system$a
  
  m <- matrix(data     = 0L,
              nrow     = length(constants),
              ncol     = length(axis),
              dimnames = list(constants, axis))
  
  x <- look(code, system)
  
  for (i in axis) { 
    m[x[i], i] <- 1L 
    }
  
  return(m)
}
